 53     /**
 54      * Normalizes an array with key-value pairs representing
 55      * a GP_Translation object.
 56      *
 57      * @since 1.0.0
 58      *
 59      * @param array $args Arguments for a GP_Translation object.
 60      * @return array Normalized arguments for a GP_Translation object.

 62     public function normalize_fields( $args ) {
 63         $args = (array) $args;
 64 
 65         if ( isset( $args['translations'] ) && is_array( $args['translations'] ) ) {
 66             foreach( range( 0, $this->get_static( 'number_of_plural_translations' ) ) as $i ) {
 67                 if ( isset( $args['translations'][ $i ] ) ) {
 68                     $args["translation_$i"] = $args['translations'][ $i ];
 69                 }
 70             }
 71             unset( $args['translations'] );
 72         }
 73         foreach( range( 0, $this->get_static( 'number_of_plural_translations' ) ) as $i ) {
 74             if ( isset( $args["translation_$i"] ) ) {
 75                 $args["translation_$i"] = $this->fix_translation( $args["translation_$i"] );
 76             }
 77         }
 78         if ( gp_array_get( $args, 'warnings' ) == array() ) {
 79             $args['warnings'] = null;
 80         }
 81         return $args;
 82     }

 92     public function fix_translation( $translation ) {
 93         // when selecting some browsers take the newlines and some don't
 94         // that's why we don't want to insert too many newlines for each ↵
 95         $translation = str_replace( "↵\n", "↵", $translation );
 96         return str_replace( '↵', "\n", $translation );
 97     }

131     public function for_translation( $project, $translation_set, $page, $filters = array(), $sort = array() ) {
132         global $wpdb;
133         $locale = GP_Locales::by_slug( $translation_set->locale );
134 
135         $join_type = 'INNER';
136 
137         $sort_bys = array('original' => 'o.singular %s', 'translation' => 't.translation_0 %s', 'priority' => 'o.priority %s, o.date_added DESC',
138             'random' => 'o.priority DESC, RAND()', 'translation_date_added' => 't.date_added %s', 'original_date_added' => 'o.date_added %s',
139             'references' => 'o.references' );
140 
141         $default_sort = get_user_option( 'gp_default_sort' );
142         if ( ! is_array( $default_sort ) ) {
143             $default_sort = array(
144                 'by'  => 'priority',
145                 'how' => 'desc'
146             );
147         }
148 
149         $sort_by = gp_array_get( $sort_bys, gp_array_get( $sort, 'by' ),  gp_array_get( $sort_bys, $default_sort['by'] ) );
150         $sort_hows = array('asc' => 'ASC', 'desc' => 'DESC', );
151         $sort_how = gp_array_get( $sort_hows, gp_array_get( $sort, 'how' ), gp_array_get( $sort_hows, $default_sort['how'] ) );
152         $collation = 'yes' === gp_array_get( $filters, 'case_sensitive' ) ? 'BINARY' : '';
153 
154         $where = array();
155         if ( gp_array_get( $filters, 'term' ) ) {
156             $like = "LIKE $collation '%" . ( esc_sql( $wpdb->esc_like( gp_array_get( $filters, 'term' ) ) ) ) . "%'";
157             $where[] = '(' . implode( ' OR ', array_map( function( $x ) use ( $like ) { return "($x $like)"; }, array( 'o.singular', 't.translation_0', 'o.plural', 't.translation_1', 'o.context', 'o.references' ) ) ) . ')';
158         }
159         if ( gp_array_get( $filters, 'before_date_added' ) ) {
160             $where[] = $wpdb->prepare( 't.date_added > %s', gp_array_get( $filters, 'before_date_added' ) );
161         }
162         if ( gp_array_get( $filters, 'translation_id' ) ) {
163             $where[] = $wpdb->prepare( 't.id = %d', gp_array_get( $filters, 'translation_id' ) );
164         }
165         if ( gp_array_get( $filters, 'original_id' ) ) {
166             $where[] = $wpdb->prepare( 'o.id = %d', gp_array_get( $filters, 'original_id' ) );
167         }
168         if ( 'yes' == gp_array_get( $filters, 'warnings' ) ) {
169             $where[] = 't.warnings IS NOT NULL';
170             $where[] = 't.warnings != ""';
171         } elseif ( 'no' == gp_array_get( $filters, 'warnings' ) ) {
172             $where[] = 't.warnings IS NULL';
173         }
174         if ( 'yes' == gp_array_get( $filters, 'with_context' ) ) {
175             $where[] = 'o.context IS NOT NULL';
176         }
177         if ( 'yes' == gp_array_get( $filters, 'with_comment' ) ) {
178             $where[] = 'o.comment IS NOT NULL AND o.comment <> ""';
179         }
180 
181         if ( gp_array_get( $filters, 'user_login' ) ) {
182             $user = get_user_by( 'login', $filters['user_login'] );
183             // do not return any entries if the user doesn't exist
184             $where[] = $wpdb->prepare( 't.user_id = %d', ($user && $user->ID) ? $user->ID : -1 );
185         }
186 
187         if ( ! GP::$permission->current_user_can( 'write', 'project', $project->id ) ) {
188             $where[] = 'o.priority > -2';
189         }
190 
191         $join_where = array();
192         $status = gp_array_get( $filters, 'status', 'current_or_waiting_or_fuzzy_or_untranslated' );
193         $statuses = explode( '_or_', $status );
194         if ( in_array( 'untranslated', $statuses ) ) {
195             if ( $statuses == array( 'untranslated' ) ) {
196                 $where[] = 't.translation_0 IS NULL';
197             }
198             $join_type = 'LEFT';
199             $join_where[] = 't.status != "rejected"';
200             $join_where[] = 't.status != "old"';
201             $statuses = array_filter( $statuses, function( $x ) { return $x != 'untranslated'; } );
202         }
203 
204         $all_statuses = $this->get_static( 'statuses' );
205         $statuses = array_filter( $statuses, function( $s ) use ( $all_statuses ) {
206             return in_array( $s, $all_statuses );
207         } );
208 
209         if ( ! empty( $statuses ) ) {
210             $statuses_where = array();
211             foreach( $statuses as $single_status ) {
212                 $statuses_where[] = $wpdb->prepare( 't.status = %s', $single_status );
213             }
214             $statuses_where = '(' . implode( ' OR ', $statuses_where ) . ')';
215             $join_where[] = $statuses_where;
216         }
217 
218         /**
219          * Filter the SQL WHERE clause to get available translations.
220          *
221          * @since 1.0.0
222          *
223          * @param array              $where           An array of where conditions.
224          * @param GP_Translation_Set $translation_set Current translation set.
225          */
226         $where = apply_filters( 'gp_for_translation_where', $where, $translation_set );
227 
228         $where = implode( ' AND ', $where );
229         if ( $where ) {
230             $where = 'AND '.$where;
231         }
232 
233         $join_where = implode( ' AND ', $join_where );
234         if ( $join_where ) {
235             $join_where = 'AND '.$join_where;
236         }
237 
238         $sql_sort = sprintf( $sort_by, $sort_how );
239 
240         $limit = $this->sql_limit_for_paging( $page, $this->per_page );
241 
242         $sql_for_translations = "
243             SELECT SQL_CALC_FOUND_ROWS t.*, o.*, t.id as id, o.id as original_id, t.status as translation_status, o.status as original_status, t.date_added as translation_added, o.date_added as original_added
244             FROM $wpdb->gp_originals as o
245             $join_type JOIN $wpdb->gp_translations AS t ON o.id = t.original_id AND t.translation_set_id = " . (int) $translation_set->id . " $join_where
246             WHERE o.project_id = " . (int) $project->id . " AND o.status LIKE '+%' $where ORDER BY $sql_sort $limit";
247         $rows = $this->many_no_map( $sql_for_translations );
248         $this->found_rows = $this->found_rows();
249         $translations = array();
250         foreach( (array)$rows as $row ) {
251             if ( $row->user_id && $this->per_page != 'no-limit' ) {
252                 $user = get_userdata( $row->user_id );
253                 if ( $user ) {
254                     $row->user_login = $user->user_login;
255                     $row->user_display_name = $user->display_name;
256                     $row->user_nicename = $user->user_nicename;
257                 }
258             } else {
259                 $row->user_login = $row->user_display_name = $row->user_nicename = '';
260             }
261             $row->translations = array();
262             for( $i = 0; $i < $locale->nplurals; $i++ ) {
263                 $row->translations[] = $row->{"translation_".$i};
264             }
265             $row->references = preg_split('/\s+/', $row->references, -1, PREG_SPLIT_NO_EMPTY);
266             $row->extracted_comments = $row->comment;
267             $row->warnings = $row->warnings? maybe_unserialize( $row->warnings ) : null;
268             unset($row->comment);
269             foreach( range( 0, $this->get_static( 'number_of_plural_translations' ) ) as $i ) {
270                 $member = "translation_$i";
271                 unset($row->$member);
272             }
273             $row->row_id = $row->original_id . ( $row->id? "-$row->id" : '' );
274             $translations[] = new Translation_Entry( (array)$row );
275         }
276         unset( $rows );
277         return $translations;
278     }

280     public function set_as_current() {
281         $result = $this->update( array('status' => 'old'),
282             array('original_id' => $this->original_id, 'translation_set_id' => $this->translation_set_id, 'status' => 'current') )
283         &&  $this->update( array('status' => 'old'),
284                 array('original_id' => $this->original_id, 'translation_set_id' => $this->translation_set_id, 'status' => 'waiting') )
285         && $this->update( array('status' => 'old'),
286             array('original_id' => $this->original_id, 'translation_set_id' => $this->translation_set_id, 'status' => 'fuzzy') )
287         && $this->save( array('status' => 'current') );
288 
289         return $result;

318     public function last_modified( $translation_set ) {
319         global $wpdb;
320 
321         $last_modified = wp_cache_get( $translation_set->id, 'translation_set_last_modified' );
322         // Cached as "" if no translations.
323         if ( "" === $last_modified ) {
324             return false;
325         } elseif ( false !== $last_modified ) {
326             return $last_modified;
327         }
328 
329         $last_modified = $wpdb->get_var( $wpdb->prepare( "SELECT date_modified FROM {$this->table} WHERE translation_set_id = %d AND status = %s ORDER BY date_modified DESC LIMIT 1", $translation_set->id, 'current' ) );
330         wp_cache_set( $translation_set->id, (string) $last_modified, 'translation_set_last_modified' );
331         return $last_modified;
332     }

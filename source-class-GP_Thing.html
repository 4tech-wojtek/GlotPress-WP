502     public function sql_condition_from_php_value( $php_value ) {
503         global $wpdb;
504         if ( is_array( $php_value ) ) {
505             return array_map( array( &$this, 'sql_condition_from_php_value' ), $php_value );
506         }
507         $operator = '=';
508         if ( is_integer( $php_value ) || ctype_digit( $php_value) )
509             $sql_value = $php_value;
510         else
511             $sql_value = "'" . esc_sql( $php_value )  ."'";
512         if ( is_null( $php_value ) ) {
513             $operator = 'IS';
514             $sql_value = 'NULL';
515         }
516         return "$operator $sql_value";
517     }

519     public function sql_from_conditions( $conditions ) {
520         if ( is_string( $conditions ) ) {
521             $conditions;
522         } elseif ( is_array( $conditions ) ) {
523             $conditions = array_map( array( &$this, 'sql_condition_from_php_value' ), $conditions );
524             $string_conditions = array();
525 
526             foreach ( $conditions as $field => $sql_condition ) {
527                 if ( is_array( $sql_condition ) ) {
528                     $string_conditions[] = '(' . implode( ' OR ', array_map( function( $cond ) use ( $field ) {
529                             return "$field $cond";
530                         }, $sql_condition ) ) . ')';
531                 } else {
532                     $string_conditions[] = "$field $sql_condition";
533                 }
534             }
535 
536             $conditions = implode( ' AND ', $string_conditions );
537         }
538         return $this->apply_default_conditions( $conditions );
539     }

617     public function set_memory_limit( $new_limit ) {
618         $current_limit     = ini_get( 'memory_limit' );
619 
620         if ( '-1' == $current_limit ) {
621             return false;
622         }
623 
624         $current_limit_int = intval( $current_limit );
625         if ( false !== strpos( $current_limit, 'G' ) ) {
626             $current_limit_int *= 1024;
627         }
628 
629         $new_limit_int = intval( $new_limit );
630         if ( false !== strpos( $new_limit, 'G' ) ) {
631             $new_limit_int *= 1024;
632         }
633 
634         if ( -1 != $current_limit && ( -1 == $new_limit || $current_limit_int < $new_limit_int ) ) {
635             ini_set( 'memory_limit', $new_limit );
636             return true;
637         }
638 
639         return false;
640     }

148     public function route() {
149         global $wp_query;
150 
151         $real_request_uri = $this->request_uri();
152         $api_request_uri = $real_request_uri;
153         $request_method = strtolower( $this->request_method() );
154         $api = gp_startswith( $real_request_uri, '/'.$this->api_prefix.'/' );
155 
156         if ( $api ) {
157             $real_request_uri = substr( $real_request_uri, strlen( $this->api_prefix ) + 1 );
158         }
159 
160         $url_path = gp_url_path( gp_url_public_root() );
161 
162         // If the request URL doesn't match our base URL, don't bother trying to match
163         if ( $url_path && ! gp_startswith( wp_unslash( $_SERVER['REQUEST_URI'] ), $url_path ) ) {
164             return;
165         }
166 
167         foreach( array( $api_request_uri, $real_request_uri ) as $request_uri ) {
168             foreach( $this->urls as $re => $func ) {
169                 foreach (array('get', 'post', 'head', 'put', 'delete') as $http_method) {
170                     if ( gp_startswith( $re, $http_method.':' ) ) {
171                         if ( $http_method != $request_method ) continue;
172                         $re = substr( $re, strlen( $http_method . ':' ));
173                         break;
174                     }
175                 }
176 
177                 if ( preg_match("@^$re$@", $request_uri, $matches ) ) {
178                     /*
179                      * WordPress will be returning a 404 status header by default for GlotPress pages
180                      * as nothing is found by WP_Query.
181                      * This overrides the status header and the `$is_404` property of WP_Query if we've matched
182                      * something here. Route controllers still can return a 404 status header.
183                      */
184                     status_header( '200' );
185                     $wp_query->is_404 = false;
186 
187                     if ( is_array( $func ) ) {
188                         list( $class, $method ) = $func;
189                         $route = new $class;
190                         $route->api = $api;
191                         $route->last_method_called = $method;
192                         $route->class_name = $class;
193                         GP::$current_route = &$route;
194                         $route->before_request();
195                         $route->request_running = true;
196                         // make sure after_request() is called even if we $this->exit_() in the request
197                         register_shutdown_function( array( &$route, 'after_request') );
198                         call_user_func_array( array( $route, $method ), array_slice( $matches, 1 ) );
199                         $route->after_request();
200                         $route->request_running = false;
201                     } else {
202                         call_user_func_array( $func, array_slice( $matches, 1 ) );
203                     }
204                     exit;
205                 }
206             }
207         }
208 
209         gp_tmpl_404();
210     }
